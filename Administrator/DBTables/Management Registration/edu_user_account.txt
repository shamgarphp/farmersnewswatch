create table edu_user_account(
id int not null auto_increment,
datee varchar(100),
timee varchar(100),
dayy varchar(100),
monthh varchar(100),
yearr varchar(100),
mgmt_code varchar(100),
first_name varchar(100),
lastname varchar(100),
mailid varchar(100),
mobile varchar(100),
password varchar(100),
status varchar(100),
edu_type varchar(100),
unique key(mgmt_code),
primary key(id));
